abcadeecfb

Q2.
Problem:
You are given a string (STR) of length N, you have to print the count of all alphabet.(using maps)

Input:
abcadeecfb 

 ans: let a='abcadeecfb'
        function countAlpha(str){
            let alphabetRk=new Map();
        
        for (let char of str){
            if (alphabetRk.has(char)){
            alphabetRk.set(char,alphabetRk.get(char)+1)
          }
          else{
             alphabetRk.set(char,1)
          }
        
        }
        console.log(alphabetRk)
        const allKeys=Array.from(alphabetRk.keys())
        for (let key of allKeys){
            console.log(key+'='+alphabetRk.get(key))
        }
        }
        countAlpha(a)
        
        Output:
        a=2
        b=2
        c=2
        d=1
        e=2
        f=1
